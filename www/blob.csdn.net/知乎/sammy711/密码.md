def enc(k,m)

 code_a=97
 kosu=26
 leng=m.length
 a=m.unpack("C*")
 b=Array.new(leng)

 for i in 0..leng-1
  if a[i] >=97 and a[i] <= 122

       if a[i-1] <97 and a[i-2] >= 97
         b[i]=a[i]-a[i-2]

       elsif a[i-1] <97 and a[i-2] <97 and a[i-3] >=97
         b[i]=a[i]-a[i-3]

       elsif a[i-1] <97 and a[i-2] <97 and a[i-3] <97 and a[i-4] >=97
         b[i]=a[i]-a[i-4]

       elsif a[i-1] <97 and a[i-2] <97 and a[i-3] <97 and a[i-4] <97 and a[i-5] >=97
         b[i]=a[i]-a[i-5]

       else
         b[i]=a[i]-a[i-1]
      end
     b[i]= (b[i] + k) % 26
     b[i]= b[i] + code_a
  else
  b[i]=a[i]
  end
 end
 b[0]=a[0]
 b[1]=(a[1]+k+1-code_a) % 26+ code_a
 c=b.pack("C*")
 return(c)
end
