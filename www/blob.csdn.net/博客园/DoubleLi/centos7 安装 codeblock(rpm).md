# centos7 安装 codeblock(rpm) - DoubleLi - 博客园






--------------------- 

1、yum -y install epel-release   

2、yum clean all && yum makecache

3、yum -y install gtk2-devel cscope binutils-devel wxGTK xterm squirrel-libs tinyxml cppcheck gamin

4、下载codeblock-17.12-1.el7.centos.x86_64.tar.xz

5、xz -d codeblock-17.12-1.el7.centos.x86_64.tar.xz 

6、tar xvf codeblock-17.12-1.el7.centos.x86_64.tar

7、rpm -ivh codeblocks*
