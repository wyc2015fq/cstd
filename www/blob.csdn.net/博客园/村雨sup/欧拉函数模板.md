返回(1~n)与n互质的数的个数

```
ll eular(ll n)
{
    ll ans = n;
    for(int i=2;i*i <= n;i++){
        if(n%i == 0){
            ans = ans - ans/i;    //p(n) = (p - p/p1)(1 - 1/p2)......
            while(n%i == 0)
                n = n/i;
        }
    }
    if(n > 1)                    //存在大于sqrt(a)的质因子
        ans = ans - ans/n;
    return ans;
}
```
