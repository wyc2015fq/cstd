**项目要求：设计一个顾客可以搜索查询账户欠款的程序。顾客ID和账户欠款提前设置好，如果欠款低于100美元，账户良好，否则显示账户欠款。**

**思路：预设10个账户及其欠款，使用冒泡排序和搜索。**

完整代码如下：

```
/*	Edited by Qin,2018.03.11
 	Filename: CusIdSear
	Function: Customer input the Id,and search the corresponding ID and credit
*/

#include <stdio.h>
 
main()
{
	int ctr;	//counter
	int idSearch;	//customer search id
	int found=0;	//1 if the customer id is found
	
	/*define ten custoner id and credit*/
	int cusId[10]={23,45,67,89,9,43,21,65,87,98};
	float cusCredit[10]={11.3,33,55,789,56,34,22.3,78,90.8,567};
	
	int tempId,inner,outer,didswap;	//for sorting
	float tempCre;
	
	/* sort the array*/
	for(outer=0;outer<9;outer++)
	{
		didswap=0;	//1 means not yet orded
		for(inner=outer;inner<10;inner++)
		{
			if(cusId[inner]<cusId[outer])
			{
				tempId=cusId[inner];	//swap both array
				tempCre=cusCredit[inner];
				cusId[inner]=cusId[outer];
				cusCredit[inner]=cusCredit[outer];
				cusId[outer]=tempId;
				cusCredit[outer]=tempCre;
				didswap=1;
			}
		 } 
		 if(didswap==0)	//quit if list is sorted
		 {
		 	break;
		 }
	}
/*interact with the user looking for credit*/
printf("\n\n***Customer Balance Lookup***\n\n");
printf("What is the customer number?   ");
scanf("%d",&idSearch);
/* now,look for the balance in the array*/
for(ctr=0;ctr<10;ctr++)
	{
		if(idSearch==cusId[ctr])
		{
			found=1;
			break;
		}	
		if(cusId[ctr]>idSearch)	//don't go too far,this is very important!
		{
			break;	//know early if id is not in list
		}
	}
	if(found)	//cleaner than if(found==1)
	{
		if(cusCredit[ctr]>100.00)
		{
			printf("\n**That customer's balance is $%.2f.\n",cusCredit[ctr]);
			printf("No credit\n");
		}
		else
		{
			printf("**The customer's credit is good!");
		}
	}
	else
		{
			printf("**The input number is wrong!");
		}
		return 0;	
}
```
