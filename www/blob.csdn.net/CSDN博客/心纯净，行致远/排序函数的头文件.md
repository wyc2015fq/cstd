在c++编程中，则是通过调用sort(a,a+n)实现排序。相应头文件是<algorithm>。

```cpp
#include<stdio.h>
#include<algorithm>
#define N 4
using namespace std;
int main()
{
	int a[N],i;
	for(i=0;i<N;i++)  
    {  
        scanf("%d",&a[i]);  
    }  
	sort(a,a+N);
	for(i=0;i<N;i++)  
    {  
        printf("%d ",a[i]);  
    }  
	return 0;
}
```
