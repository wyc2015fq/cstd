 push ax
 push dx
 push cx

 cmp al,0
 jz ok1
 cmp al,97

 cmp al,122

 jmp s

 pop dx
 pop ax

 mov ch,0

 mov bl,ds:[si]
        mov es:[di],bx

        add di,2
