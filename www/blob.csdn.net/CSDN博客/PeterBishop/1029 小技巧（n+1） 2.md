int a[MAXN];
int main()
{
	//freopen("out.txt","w",stdout);
	freopen("in.txt","r",stdin);
    int N;
	while(cin>>N){
		for(int i=1;i<=N;i++)
		{
			cin>>a[i];
		}
		sort(a,a+N);
		cout<<a[(N+1)/2]<<endl;
	}
    
    
    return 0;
}
