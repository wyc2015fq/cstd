using namespace std;
int dp[100001][12];
int main()
{
	freopen("in.txt","r",stdin);
	int n;
    while(cin>>n&&n){
		memset(dp,0,sizeof(dp));
		int t,x,m=0;
		for(int i=1;i<=n;i++){
			scanf("%d %d",&x,&t);
			dp[t][x+1]++;
			m=max(m,t);
		}
		int ans=0;
		for(int i=m-1;i>=0;i--){
			for(int j=11;j>=1;j--){
				dp[i][j]+=max(dp[i+1][j],max(dp[i+1][j+1],dp[i+1][j-1]));
				ans=max(ans,dp[i][j]);
			}
		}
		cout<<ans<<endl;

	}
    
    
    
    return 0;
}
