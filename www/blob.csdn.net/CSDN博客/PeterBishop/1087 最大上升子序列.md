#include <iostream>

#include <cstdlib>
#include <vector>
#include <queue>

#include <stack>
#include <string>

using namespace std;

int main()
{
	freopen("in.txt","r",stdin);
	int n;
	while(cin>>n){
		if(!n)
			break;
		int a[1010];
		a[0]=0;
		for(int i=1;i<=n;i++)
			cin>>a[i];
		int dp[1010];
		int max=0;
		memset(dp,0,sizeof(dp));
		dp[1]=a[1];
		for(int i=1;i<=n;i++){
			for(int j=1;j<i;j++){
				if(a[i]>a[j]&&dp[j]+a[i]>dp[i]){
					dp[i]=dp[j]+a[i];
				}
			}

			if(max<dp[i])
				max=dp[i];

		}
		cout<<max<<endl;
	}
    return 0;
}
