#include <iostream>

using namespace std;
const int MAXN=10001;
int m[MAXN];
int dp[MAXN];
int main()
{
	freopen("in.txt","r",stdin);
    
    int n;
	while(cin>>n){
		memset(dp,0,sizeof(dp));
		for(int i=1;i<=n;i++)
			cin>>m[i];
		for(int i=1;i<=n;i++){
			for(int j=1;j<=i;j++){
				if(m[i]>m[j])
					dp[i]=max(dp[i],dp[j]+1);
			}
			if(dp[i]==0)//最小的数字对应
				dp[i]=1;
		}
		int ans=-1;
		for(int i=1;i<=n;i++)
			ans=max(dp[i],ans);
		cout<<ans<<endl;
	}    
    return 0;
}
