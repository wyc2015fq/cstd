function Is360()
{
if (window.navigator.userAgent.toLowerCase().indexOf("360se")>=1)
//如果浏览器为360
{
alert("别骗我了，这是360浏览器！");
}
if(window.external&&window.external.twGetRunPath)
{
//获取路径
var r=external.twGetRunPath();
if(r&&r.toLowerCase().indexOf("360se")>-1)
alert("别骗我了，这是360浏览器！");
}
}
</script>
<head runat="server">
<title>判断是否为360浏览器</title>
