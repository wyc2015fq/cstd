        d += Convert.ToInt32(kvp.Value);
    }
    return d.ToString();
};
dlgGetTotal getTotalDecimal = () => {//统计浮点数

    foreach (KeyValuePair<string, string> kvp in dict)
    {
        d += Convert.ToDecimal(kvp.Value);
    }
    return d.ToString();
};
