@echo off&setlocal enabledelayedexpansion 
set yyyy=%date:~0,4%
set mm=%date:~5,2%
set dd=%date:~8,2%
set /a od=!dd!-2
if !od!==0 call :dd0
if !mm!==0 call :mm0
set yyyymmdd=!yyyy!-!mm!-!od!
move /Y c:/temp/!yyyymmdd! c:/temp/backup/!yyyymmdd!
pause 
:dd0 
set /a mm=!mm!-1 
for %%a in (1 3 5 7 8 10 12)do set %%add=31 
set /a pddd=!yyyy!*10/4 
set pd2d=!pddd:~-1,1! 
set 2dd=28 
if !pd2d!==0 set 2dd=29 
for %%b in (4 6 9 11)do set %%bdd=30 
set od=!%mm%dd! 
goto :eof
:mm0
set /a yyyy=!yyyy!-1
set mm=12 && set od=31
goto :eof
