ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
    int na = 0, nb = 0;
    ListNode *headAA = headA;
    ListNode *headBB = headB;
    while (headAA) {
        headAA = headAA->next;
        na++;
    }
    while (headBB) {
        headBB = headBB->next;
        nb++;
    }
    int n = na -nb;
    if (na > nb) {
        while (n--) {
            headA = headA->next;
        }
    }
    else if(nb>na){
        n = -n;
        while (n--) {
            headB = headB->next;
        }
    }
    while (headA != headB) {
        headA = headA->next;
        headB = headB->next;
    }
    return headA;
}
