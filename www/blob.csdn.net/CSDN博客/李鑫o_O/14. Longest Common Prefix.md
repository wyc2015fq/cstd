string longestCommonPrefix(vector<string>& strs) {
    string res = "";
    if(strs.size()==0) return res;
    for (int j = 0; j < strs[0].size(); j++){
        char temp = strs[0][j];
        for (int i = 0; i < strs.size(); i++) 
            if (j > strs[i].size() || temp != strs[i][j])
                return res;
        res += temp;
    }
    return res;
}
