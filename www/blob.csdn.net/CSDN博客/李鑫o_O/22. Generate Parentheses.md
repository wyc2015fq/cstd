void help(vector<string> &res,string s,int left, int right) {
    {
        if (right == 0) {
            res.push_back(s);
            return;
        }
        if (left) help(res, s + "(", left-1, right);
        if(right>left) help(res, s + ")", left, right-1);
    }
}
vector<string> generateParenthesis(int n) {
    vector<string> res;
    string s = "";
    help(res, s, n, n);
    return res;
}
