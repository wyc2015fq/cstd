int removeDuplicates(vector<int>& nums) {
    int n = nums.size();
    int res = 0;
    for (int i = 0; i < n; i++) {
        if (i + 1 < n&&nums[i] == nums[i + 1]) {
            res++;
        }
    }
    unique(nums.begin(), nums.end());
    return n - res;
}

int removeDuplicates(vector<int>& nums) {
    int n = nums.size();

    for (int i = 0; i < nums.size(); i++) {
        if (i + 1 < nums.size() &&nums[i] == nums[i + 1]) {
            nums.erase(nums.begin()+i);
            i--;
        }
    }
