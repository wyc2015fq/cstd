int reverse(int x) {
    long res = 0;
    int flag = 1;
    if (x < 0) {
        flag = -1;
        x = -x;
    }
    while (x > 0) {
        res = res * 10 + x % 10;
        x = x / 10;
    }
    if (res > INT_MAX) return 0;
    return res*flag;
}
