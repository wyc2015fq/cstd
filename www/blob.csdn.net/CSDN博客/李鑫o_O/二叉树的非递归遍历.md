### Pre Order Traverse



```cpp
vector<int> preorderTraversal(TreeNode* root) {
	vector<int> res;
	std::stack<TreeNode*> temp;
	while (root || !temp.empty()) {
		while (root) {
			temp.push(root);
			res.push_back(root->val);
			root = root->left;
		}
		root = temp.top();
		temp.pop();
		root = root->right;
	}
	return res;
}
```



### In Order Traverse

```cpp
vector<int> inorderTraversal(TreeNode* root) {
	vector<int> res;
	std::stack<TreeNode*> temp;
	while (root || !temp.empty()) {
		while (root) {
			temp.push(root);
			root = root->left;
		}
		root = temp.top();
		temp.pop();
		res.push_back(root->val);
		root = root->right;
	}
	return res;
}
```




### Post Order Traverse



```cpp
vector<int> postorderTraversal(TreeNode* root) {
	vector<int> res;
	std::stack<TreeNode*> temp;
	while (root || !temp.empty()) {
		while (root) {
			temp.push(root);
			res.insert(res.begin(),root->val);
			root = root->right;
		}
		root = temp.top();
		temp.pop();
		root = root->left;
	}
	return res;
}
```
