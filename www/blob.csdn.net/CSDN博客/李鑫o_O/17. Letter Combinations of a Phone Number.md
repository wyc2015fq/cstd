vector<string> letterCombinations(string digits) {
    int n = digits.size();
    if (n == 0) return vector<string>();
    vector<vector<string>> res(n);
    vector<vector<string>> temp = {
        {" "},
        { "#" },
        { "a","b","c" },
        { "d","e","f" },
        { "g","h","i" },
        { "j","k","l" },
        { "m","n","o" },
        { "p","q","r","s" },
        { "t","u","v" },
        { "w","x","y","z" }
    };
    for (int i = 0; i < n; i++) {
        for (auto letter : temp[digits[i] - '0']) {
            if (i == 0) {
                res[i].push_back(letter);
                continue;
            }
            for (auto anwser : res[i-1]) {
                res[i].push_back(anwser + letter);
            }
        }
    }
    return res[n - 1];
}
