int romanToInt(string s) {
    map<char, int> T = { { 'I' , 1 },
                         { 'V' , 5 },
                         { 'X' , 10 },
                         { 'L' , 50 },
                         { 'C' , 100 },
                         { 'D' , 500 },
                         { 'M' , 1000 } };
    int sum = T[s.back()];
    for (int i = 0; i < s.size() - 1; i++) {
        if (T[s[i]] >= T[s[i + 1]]) sum += T[s[i]];
        else sum -= T[s[i]];
    }
    return sum;
}
