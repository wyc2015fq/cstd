int lengthOfLastWord(string s) {
    while (*(s.end() - 1) == ' ')
        s.erase(s.end() - 1);
    if (s.size() == 0) return 0;
    for (int i = s.size() - 1; i >= 0; i--) {
        if (s[i] == ' ') return s.size() - i - 1;
    }
    return s.size();
}
