把一个偶数拆成两个不同素数的和，有几种拆法呢？ 

  Input 

  输入包含一些正的偶数，其值不会超过10000，个数不会超过500，若遇0，则结束。 

  Output 

  对应每个偶数，输出其拆成不同素数的个数，每个结果占一行。 

  Sample Input 

  30 

  26 

  0 

  Sample Output 

  3 

  2
代码：

```cpp
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
const int MAXN=10050;
bool prime[MAXN];
void Init(){
    memset(prime,true,sizeof(prime));
    prime[0]=false;
    prime[1]=false;
    for(int i=2;i<=MAXN;i++){
        if(prime[i]){
            if(i*i>=MAXN){
                continue;
            }
            for(int j=i*i;j<=MAXN;j+=i){
                prime[j]=false;
            }
        }
    }
}
int main(void){
    Init();
    int n;
    while(~scanf("%d",&n)){
        if(n==0){
            break;
        }
        int ans=0;
        for(int i=2;i<n/2;i++){
            if(prime[i] && prime[n-i]){
                //printf("%d %d\n",i,n-i);
                ans++;
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
```
