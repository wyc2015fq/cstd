```cpp
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
int main(void){
    //物品种类
    int n=5;
    //背包可装重量
    int c=10;
    //每一种物品重量
    int w[5]={2,3,4,2,5};
    //每一种物品价值
    int v[5]={4,3,3,4,5};
    //dp[i][j]表示前i个物品背包可用容量（重量）为j时的最大价值
    int dp[6][11];
    memset(dp,0,sizeof(dp));
    for(int i=1;i<=n;i++){
       for(int j=0;j<=c;j++){
           //可以放（但不一定放）
           if(j>=w[i]){
               //放的话需要先腾出空间，所以先等于dp[i-1][j-w[i]]保证有空间放，然后再放（+v[i]）
               dp[i][j]=max(dp[i-1][j],dp[i-1][j-w[i]]+v[i]);
           }
           //不可以放
           else{
               dp[i][j]=dp[i-1][j];
           }
       } 
    }
    printf("   j0 j1 j2 j3 j4 j5 j6 j7 j8 j9 j10\n");
    for(int i=0;i<6;i++){
        printf("i%d ",i);
        for(int j=0;j<11;j++){
            printf("%2d ",dp[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
