# 【题目描述】

一天，处女座在牛客算法群里发了一句“我好强啊”，引起无数的复读，可是处女座发现复读之后变成了“处女座好强啊”。处女座经过调查发现群里的复读机都是失真的复读机，会固定的产生两个错误。一个错误可以是下面的形式之一：

1.将任意一个小写字母替换成另外一个小写字母

2.在任意位置添加一个小写字母

3.删除任意一个字母

处女座现在在群里发了一句话，他收到了一个回应，他想知道这是不是一个复读机。

# 【输入描述】

两行

第一行是处女座说的话s

第二行是收到的回应t

s和t只由小写字母构成且长度小于100

# 【输出描述】

如果这可能是一个复读机输出”YES”，否则输出”NO”

# 【样例】

示例1

输入

abc

abcde

输出

YES

说明

abc->abcd->abcde

示例2

输入

abcde

abcde

输出

YES

说明

abcde->abcdd->abcde


思路：大模拟。。具体看代码

#include<iostream>

using namespace std;
int main(){

    string s,t;
    cin>>s>>t;
    int lenS=s.size();
    int lenT=t.size();

    if(abs(lenS-lenT)>2){//两字符串长度差值大于2
        printf("NO\n");
    }
    else if(abs(lenS-lenT)==2){//两字符串长度差值等于2，一定是两次添加或两次删除
        if(lenS>lenT){//交换两字符串保证s是长度短的，就不用分情况讨论是进行两次添加还是两次删除
            swap(s,t);
            swap(lenS,lenT);
        }

        int num=0;//不同的字母个数
        for(int i=0,j=0;i<lenS;i++,j++){
            if(s[i]!=t[j]){//若s的第i个与t的第j个不同，说明进行了修改
                num++;
                i--;
            }
            if(num>2||j>=lenT){//不同的字母个数不能大于2
                printf("NO\n");
                return 0;
            }
        }
        printf("YES\n");
    }
    else if(abs(lenS-lenT)==1){//两字符串长度差值等于1，一定是替换+删除、替换+增加、删除+替换、删除+增加中的一种
        if(lenS>lenT){//交换两字符串保证s是长度短的，就不用分情况讨论是替换+删除、替换+增加、删除+替换、删除+增加，变为只需讨论是替换+删除/增加或删除/增加+替换
            swap(s,t);
            swap(lenS,lenT);
        }

        int num=0;//不同的字母个数
        bool pos=true;//第一个不同的字母
        bool flag=true;
        for(int i=0,j=0;i<lenS;i++,j++){
            if(s[i]!=t[j]){//若s的第i个与t的第j个不同，说明进行了修改
                if(pos){//第一个不同的字母表示进行后移
                    i--;
                    pos=false;
                }
                num++;
            }
            if(num>2||j>=lenT){//不同的字母个数不能大于2
                flag=false;
                break;
            }
        }

        if(flag&&num<=2)
            printf("YES\n");
        else{
            num=0;
            for(int i=0,j=0;i<lenS;i++,j++){
                if(s[i]!=t[j]){//若s的第i个与t的第j个不同，说明进行了修改
                    num++;
                    if(num==2)//第二个不同的字母表示进行后移
                        i--;
                }
                if(num>2){
                    printf("NO\n");
                    return 0;
                }
            }
            printf("YES\n");
        }

    }
    else if(abs(lenS-lenT)==0){//两字符串长度差值等于0，一定是替换+替换、删除+增加、增加+删除中的一种
        int num=0;//不同的字母个数

        //替换+替换
        for(int i=0;i<lenS;i++)//记录两字符串中不相等的个数
            if(s[i]==t[i])
                num++;

        //删除+增加、增加+删除
        if(lenS-num<=2)//改变的元素小于等于2个，说明进行了两次替换
            printf("YES\n");
        else{
            //删除+增加
            num=0;//不同的字母个数
            bool flag=true;
            for(int i=0,j=0;i<lenS;i++,j++){
                if(s[i]!=t[j]){
                    num++;
                    if(num==1)
                        i--;
                    if(num==2)
                        j--;
                }
                if(num>2||j>=lenT){
                    flag=false;
                    break;
                }
            }

            //增加+删除
            if(flag)
                printf("YES\n");
            else{
                num=0;//不同的字母个数
                for(int i=0,j=0;i<lenS;i++,j++){
                    if(s[i]!=t[j]){
                        num++;
                        if(num==1)
                            j--;
                        if(num==2)
                            i--;
                    }
                    if(num>2||j>=lenT){
                        printf("NO\n");
                        return 0;
                    }
                }
                printf("YES\n");
            }
        }
    }
	return 0;
}
