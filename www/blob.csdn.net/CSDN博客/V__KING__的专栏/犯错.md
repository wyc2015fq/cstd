# 犯错
- 给结构体赋值的时候,给数组赋值的时候不需要中括号[]，有[]是有问题的。

```
struct cmd_buf
{
    UINT8   command_type;                   // 指令类型
    UINT16  param_length;                   // 指令长度
    UINT8   param_buff[100];        // 指令的参数BUF
};

// use to test
struct cmd_buf cmd = {
    .command_type    = 0x6d,
    .param_length    = 23,
    .param_buff      = {
        0x00 ,0x00, 0x00 ,0x00 ,
        0x00,           
        0xa5,0xc3  ,           
        0x73,0x76 ,0x6e ,0x32 ,0x34 ,0x32  ,           
        0x00,0x00 ,0x06 ,0xd5 ,           
        0x00,0x03 ,0x6a ,0x53 ,
        0x47,0x22
    },
};
```
