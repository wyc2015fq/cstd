必败点：斐波拉契数！
```cpp
#include<iostream>
using namespace std;
long long d[50];
int main()
{
    d[1]=1,d[2]=2;
    for(int i=3;i<50;i++)
      d[i]=d[i-1]+d[i-2];
    int n;
    while(cin>>n&&n)
    {
        int blag=0;
        for(int i=1;d[i]<=n;i++)
          if(n==d[i])
            blag=1;
        if(blag)
          cout<<"Second win"<<endl;
        else
          cout<<"First win"<<endl;
    }
}
```
