queue<int> q;  
vector<int> vec[maxn];  //邻接表表示点与点之间的关系 
int n,m,Indeg[maxn];  //表示入度的个数 
bool topsort()
{
	for(int i=1;i<=n;i++)
	  if(!Indeg[i])
	    q.push(i);
	int cnt=0;
	while(!q.empty())
	{
		int now=q.front();
		q.pop();
		cnt++;
		for(int i=0;i<vec[now].size();i++)
		  if(--Indeg[vec[now][i]]==0)
		    q.push(vec[now][i]);
	}
	return cnt==n;
}
