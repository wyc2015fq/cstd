ll mult_mod(ll a,ll n)
{
    ll ans=0;
    a%=mod,n%=mod;
    while(n)
    {
        if(n&1)
          ans=(ans+a)%mod;
        a=(a+a)%mod;
        n>>=1;
    } 
    return ans%mod;
}
ll quick_mod(ll a,ll n)
{
    ll ans=1;
    while(n)
    {
        if(n&1)
          ans=mul_mod(ans,a);
        a=mul(a,a);
        n>>=1;
    }
    return ans;
}
