裸的错排问题。
```cpp
#include<iostream>
using namespace std;
#define mod 1000000007 
long long d[1005];
int main()
{
    int t,n;
    d[1]=0,d[2]=1;
    for(int i=3;i<=100;i++)
      d[i]=(i-1)*(d[i-1]+d[i-2])%mod;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<d[n]<<endl;
    }
    return 0;
}
```
