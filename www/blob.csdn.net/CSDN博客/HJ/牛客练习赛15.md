A题 吉姆的运算式
解题思路：取串中最后一个数值。
```cpp
#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char s[105];
    while(cin>>s)
    {
        int len=strlen(s)-1,i=len;
        long long res=0,ans=1;
        while(s[i]==')')
          i--;
        while(s[i]!=','&&s[i]!='-'&&i>=0&&s[i]!='(')
        {
            res+=(s[i]-'0')*ans;
            ans*=10;
            i--;
        }
        if(s[i]=='-')
          res=-res;
        cout<<res<<endl;
    }
    return 0;
}
```
B题 沃老师学生的成绩
解题思路：将成绩去掉后导0重新排序，如果成绩相同按字典序输出。
```cpp
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
struct node{
    string name,score1,score2;
}s[200015];
bool cmp(const node &a,const node &b)
{
    if(a.score1==b.score1)
      return a.name<b.name;
    return a.score1>b.score1;
}
int main()
{
    int n;
    while(cin>>n)
    {
        for(int i=0;i<n;i++)
        {
            cin>>s[i].name>>s[i].score1;
            s[i].score2=s[i].score1;
            int j=s[i].score1.length();
            while(s[i].score1[j-1]=='0')
              j--;          
            s[i].score1.resize(j);
        }
        sort(s,s+n,cmp);
        for(int i=0;i<n;i++)
          cout<<s[i].name<<" "<<s[i].score2<<endl;
    }
    return 0;
}
```
