统计每个字母出现的次数，处理一下就好了。
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define maxn 35
int a[30],b[maxn+5];
char s[maxn+5]; 
int main()
{
    int n,tt=0;
    while(~scanf("%d",&n))
    {
        int cnt=0;
        while(n--)
        {
            int blag=0,k=0;
            memset(a,0,sizeof(a));
            memset(b,0,sizeof(b));
            memset(s,0,sizeof(s));
            scanf("%s",s);
            for(int i=0;s[i]!='\0';i++)
              a[s[i]-'a']++;
            for(int i=0;i<26;i++)
              if(a[i])
              {
                  b[a[i]]++;           //字母i+'a'出现次数为a[i]的种数 
                  k++;
              }
            if(k==1)
              continue;
            for(int i=0;i<maxn;i++)
              if(b[i]>1)
                blag=1;
            if(blag==0)
              cnt++;
        }
        printf("Case %d: %d\n",++tt,cnt);
    }
    return 0;
}
```
