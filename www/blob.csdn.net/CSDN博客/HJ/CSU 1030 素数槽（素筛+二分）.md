解题思路： 
首先素筛打表一下，然后二分查找出来即可；
```cpp
#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<cstdio>
const int maxn=1300000;
int a[maxn+10]; 
int m;
void Init()
{
    memset(a,0,sizeof(a));
    for(int i=2;i<=sqrt(maxn+0.5);i++)
      if(a[i]==0)
        for(int j=i*i;j<=maxn;j+=i)
          a[j]=1;
    for(int i=2;i<=maxn;i++)
      if(a[i]==0)
        a[m++]=i;
}
using namespace std;
int main()
{
    Init();
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int k=lower_bound(a,a+m,n)-a;
        printf("%d\n",a[k]==n?0:a[k]-a[k-1]);
    }
    return 0;
}
```
