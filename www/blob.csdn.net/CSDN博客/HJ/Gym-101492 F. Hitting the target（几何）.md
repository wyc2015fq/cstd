刚开始的时候按照矩形来做，wa几发之后才知道需要按照圆来做，需要计算一下dis距离。
```cpp
#include<iostream>
#include<cmath>
#include<cstdlib>
using namespace std;
double dis(int x, int y)
{
    double ans = sqrt(1.0*x*x + 1.0*y*y);
    return ans;
}
int main()
{
    int r;
    while(cin>>r)
    {
        int ans=0,x,y;
        for(int i=0;i<3;i++)
        {
            cin>>x>>y;
            double dist = dis(x,y);
            if(dist > 10*r) continue;
            int temp=0 ;
            for(int i=0;  i < 10; i++)
            {
                if(dist >= i*r && dist <= (i+1)*r)
                {
                    temp = i;
                    break;
                }
            }
            ans += (10 - temp);
        } 
        cout<<ans<<endl;
    }
    return 0;
}
```
