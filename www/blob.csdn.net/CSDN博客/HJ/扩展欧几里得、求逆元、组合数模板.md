ll exgcd(ll a,ll b,ll &x,ll &y)
{
    if(b==0)
    {
        x=1,y=0;
        return a;
    }
    ll res=exgcd(b,a%b,y,x);
    y-=a/b*x;
    return res;
}
ll Inv(ll a)
{
    ll d,x,y;
    d=exgcd(a,mod,x,y);
    if(d==1)
      return (x%mod+mod)%mod;
    return -1;
} 
ll C(ll n,ll m)
{
    ll ans1=1,ans2=1;
    for(int i=n,j=1;j<=m;j++,i--)
    {
        ans1=ans1*i%mod;
        ans2=ans2*j%mod;
    }
    return (ll)(ans1*Inv(ans2)%mod);
}
