分类枚举情况就行了，分成四个板块。
```cpp
#include<iostream>
using namespace std;
struct point{
    int x,y;
};
int main()
{
    point a,b;
    while(cin>>a.x>>a.y>>b.x>>b.y)
    {
        int blag;
        if(a.x<=b.x&&a.y<=b.y)
          blag=1;
        else if(a.x>b.x&&a.y>b.y)
          blag=0;
        else 
        {
            int ans1=max(b.x,b.y);
            int ans2=a.x+a.y;
            if(ans1>=ans2)
              blag=1;
            else
              blag=0; 
        }
        if(blag)
          cout<<"Polycarp"<<endl;
        else
          cout<<"Vasiliy"<<endl;
    }
    return 0;
}
```
