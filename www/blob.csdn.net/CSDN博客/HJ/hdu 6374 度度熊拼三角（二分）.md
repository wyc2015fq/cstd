解题思路：
选择任意两个数，二分查找第三个满足条件的数即可。
```cpp
#include<iostream>
#include<algorithm>
#include<map>
#include<vector>
#include<set>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
#define mod 1000000007
typedef long long ll;
const int maxn=1e5+10;
int a[maxn];
int main()
{
    int n;
    while(cin>>n)
    {
        for(int i=0;i<n;i++)
          cin>>a[i];
        sort(a,a+n);
        int ans=0;
        for(int i=0;i<n;i++)
          for(int j=i+1;j<n;j++)
          {
              int x=a[i],y=a[j];
              int k=upper_bound(a,a+n,x+y)-a;
              if(x+y-a[k-1]>0&&k-1>j&&k-1>i)
                ans=max(ans,x+y+a[k-1]);
          }
        if(ans)
          cout<<ans<<endl;
        else
          cout<<-1<<endl;
    }
    return 0;
}
```
