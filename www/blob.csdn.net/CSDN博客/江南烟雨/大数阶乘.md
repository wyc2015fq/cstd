## 大数阶乘
时间限制：3000 ms  |  内存限制：65535 KB 
难度：3
- 描述 
我们都知道如何计算一个数的阶乘，可是，如果这个数很大呢，我们该如何去计算它并输出它？ - 输入 
输入一个整数m(0<m<=5000) - 输出 
输出m的阶乘，并在输出结束之后输入一个换行符 - 样例输入 
50- 样例输出 
30414093201713378043612608166064768844377641568960512000000000000

#include<stdio.h>
#define MAXLEN 100000
int Factorial(int n,int a[]){
	int sum,i,j,temp;
	sum=0;//最高位下标；
	a[0]=1;
	for(i=1;i<MAXLEN;i++){
		a[i]=0;
	}
	for(i=2;i<=n;i++){
		for(j=0;j<=sum;j++){
			a[j]*=i;
		}
		for(j=0;j<sum;j++){
			temp=a[j];
			a[j]=temp%10;//本位数值；
			a[j+1]+=temp/10;//进位；
		}
		while(a[sum]>9){
			temp=a[sum];
			a[sum]=temp%10;
			sum++;
			a[sum]=temp/10;
		}
	}
	//printf("%d/n",a[sum]);
	return sum;
}
int main(){
	int n,a[MAXLEN],sum,i;
	while(scanf("%d",&n)!=EOF){
		if(n<=0||n>5000){
			break;
		} 
		sum=Factorial(n,a);
		for(i=sum;i>=0;i--){
			printf("%d",a[i]);
		}
		printf("/n");
	}
}
