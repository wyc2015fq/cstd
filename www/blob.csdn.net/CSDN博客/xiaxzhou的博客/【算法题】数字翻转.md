> 
对于一个整数X，定义操作rev(X)为将X按数位翻转过来，并且去除掉前导0。例如: 

  如果 X = 123，则rev(X) = 321; 

  如果 X = 100，则rev(X) = 1. 

  现在给出整数x和y,要求rev(rev(x) + rev(y))为多少？ 
**输入描述:**

  输入为一行，x、y(1 ≤ x、y ≤ 1000)，以空格隔开。

**输出描述:**

  输出rev(rev(x) + rev(y))的值

**输入例子:**

  123 100

**输出例子:**

  223

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <numeric>
#include<algorithm>
using namespace std;
#define debug 0

int func(int x, int y)
{
    int tmp;

    int rex(0), rey(0);
    while (x != 0)
    {
        tmp = x % 10;
        x = x / 10;
        rex = rex * 10 + tmp;
    }
    while (y != 0)
    {
        tmp = y % 10;
        y = y / 10;
        rey = rey * 10 + tmp;
    }
    int rez = rex + rey;
    int z(0);
    while (rez != 0)
    {
        tmp = rez % 10;
        rez = rez / 10;
        z = z * 10 + tmp;
    }
    return z;
}

int main()
{
    int x, y;
    if (debug)
    {
        x = 123;
        y = 100;

    }
    else
    {
        cin >> x >> y;
    }
    cout << func(x,y) << endl;

      return 0;
}
```
