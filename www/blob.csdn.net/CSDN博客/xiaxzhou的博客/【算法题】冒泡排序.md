void func(int* array,int len)
{
    if (array == NULL || len<=1)
    {
        return;
    }
    for (auto i = 0; i < len;++i)
    {
        for (auto j = len-1; j > i;--j)
        {
            if (array[j]<array[j-1])
            {
                std::swap(array[j],array[j-1]);
            }
        }
    }
}
