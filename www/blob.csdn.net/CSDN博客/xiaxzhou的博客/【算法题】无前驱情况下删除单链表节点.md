分两种情况考虑：无法删除的**尾节点**；删除非尾节点


```
bool Deletnode(ListNode * pnode)
{
    if (pnode==NULL||pnode->next == NULL)
    {
        return false;
    }
    ListNode* ptmp = pnode->next;
    pnode->value = ptmp->value;
    pnode->next = ptmp->next;

    delete ptmp;
    return true;
}
```
