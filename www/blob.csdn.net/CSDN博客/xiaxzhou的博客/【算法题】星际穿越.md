航天飞行器是一项复杂而又精密的仪器，飞行器的损耗主要集中在发射和降落的过程，科学家根据实验数据估计，如果在发射过程中，产生了 x 程度的损耗，那么在降落的过程中就会产生 x2 程度的损耗，如果飞船的总损耗超过了它的耐久度，飞行器就会爆炸坠毁。问一艘耐久度为 h 的飞行器，假设在飞行过程中不产生损耗，那么为了保证其可以安全的到达目的地，只考虑整数解，至多发射过程中可以承受多少程度的损耗？ 

**输入描述:**

  每个输入包含一个测试用例。每个测试用例包含一行一个整数 h （1 <= h <= 10^18）。

**输出描述:**

  输出一行一个整数表示结果。

**输入例子:**

  10

**输出例子:**

  2

**查找上界**

**二分查找**

注意：使用long long 防止溢出

```cpp
#include <iostream>
#include <numeric>
#include<algorithm>
#include <string>
#include<hash_map>
#include <set>
using namespace std;
//#define debug_

long long func(long long n)
{
    long long left = 0;
    long long right = sqrt(n)+1;
    long long mid = left + (right - left) >> 1;

    long long tmp;
    while (left +1 < right)
    {
        mid = left + ((right - left) >> 1);

        tmp = mid + mid*mid;
        if (tmp < n)
        {
            left = mid;
        }
        if (tmp>n)
        {
            right = mid;
        }
        if (tmp == n)
        {
            return mid;
        }
    }

    tmp = right + right*right;

    if (tmp<=n)
    {
        return right;
    }
    else
    {
        return left;
    }
}

int main()
{
    long long n;
#ifdef debug_
    n = 10;

#else

    cin>>n;
#endif
    cout << func(n) << endl;

    return 0;
}
```
