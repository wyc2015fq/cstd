void func(int* array,int len)
{
    if (array == NULL || len<=1)
    {
        return;
    }
    for (auto i = 1; i < len;++i)
    {
        for (auto j = i; j >= 1;--j)
        {
            if (array[j]<array[j-1])
            {
                std::swap(array[j],array[j-1]);
            }
            else
            {
                break;
            }
        }
    }
}
