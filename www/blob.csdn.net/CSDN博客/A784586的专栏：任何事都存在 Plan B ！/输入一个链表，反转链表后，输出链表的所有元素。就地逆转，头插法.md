    public ListNode ReverseList(ListNode head) {
    	if(head==null){
    		return null;
    	}
    	if(head.next==null){
    		return head;
    		}
    	ListNode newhead=head;
    	ListNode p=head.next,q=head.next;
    	
    	newhead.next=null;
    	
    	while(p.next!=null){
    		q=p.next;
    		p.next=newhead;
    		newhead=p;
    		p=q;
    	}
    	p.next=newhead;
    	newhead=p;
    	
    	return newhead;
    }
}
