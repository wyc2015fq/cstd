Tarjan 缩点


#define maxn 100010
vector<int>edge[maxn];
int in[maxn];
int color[maxn];  //第i个节点属于哪个联通块
int stack[maxn];
int vis[maxn];
int dfn[maxn];
int low[maxn];
int n,m,top,cnt,sig;
void init()
{
    top=0;cnt=1;sig=0;
    memset(in,0,sizeof(in));
    memset(color,0,sizeof(color));
    memset(stack,0,sizeof(stack));
    memset(low,0,sizeof(low));
    memset(dfn,0,sizeof(dfn));
    memset(vis,0,sizeof(vis));
    for(int i=1;i<=n;i++)edge[i].clear();
}
void Tarjan(int u)
{
    vis[u]=1;
    low[u]=dfn[u]=cnt++;
    stack[++top]=u;
    for(int i=0;i<edge[u].size();i++)
    {
        int v=edge[u][i];
        if(vis[v]==0)Tarjan(v);
        if(vis[v]==1)low[u]=min(low[u],low[v]);
    }
    if(dfn[u]==low[u])
    {
        sig++;
        do
        {
            color[stack[top]]=sig;
            vis[stack[top]]=-1;
        }
        while(stack[top--]!=u);
    }
}
void Slove()
{
    
    for(int i=1;i<=n;i++)
    {
        if(vis[i]==0)
        {
            Tarjan(i);
        }
    }
}
int main()
{
	int t;
	scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&m);
        init();
        for(int i=0;i<m;i++)
        {
            int x,y;
            scanf("%d%d",&x,&y);
            edge[x].push_back(y);
        }
        Slove();
    }
}
