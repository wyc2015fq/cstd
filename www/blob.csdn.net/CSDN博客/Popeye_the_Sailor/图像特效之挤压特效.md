## 图像特效之挤压特效
### 1.关键代码
```cpp
int Pinch(IMAGE_TYPE *BMP24_img,int degree)
{
	DWORD width,height,x,y,bfsize;
	WORD  biBitCount = 3;
	T_U8 *dst_data,*bmp,*img_data;
	T_U32 line_byte,src_Imgindex,dst_Imgindex;
	int Xoffset,Yoffset,MidX,MidY,X,Y;
	double radius,radian;
	BITMAPFILEHEADER bf;
	BITMAPINFOHEADER bi;
	
	FILE *Pinch_fp = fopen("Pinch.bmp","wb");
	
	if(NULL == Pinch_fp)
	{
		printf("Can't open Pinch_fp.bmp\n");
		return -1;
	}
	
	if (degree < 1) degree = 1;
	if (degree > 32) degree = 32;
	memset(&bf, 0, sizeof(bf));
	memset(&bi, 0, sizeof(bi));
	
	bmp = BMP24_img;
	memcpy(&bf,bmp,14);
	memcpy(&bi,&bmp[14],40);
	
	fwrite(&bf,sizeof(BITMAPFILEHEADER),1,Pinch_fp);
	fwrite(&bi,sizeof(BITMAPINFOHEADER),1,Pinch_fp);
	
	height = bi.biHeight;
	width  = bi.biWidth;
	bfsize = bf.bfSize;
	biBitCount = bi.biBitCount;
	line_byte =  (width*3+3) & ~0x3;
	img_data = bmp+54;
	
	dst_data = (T_U8 *)malloc(line_byte*height);
	if (NULL == dst_data)
	{
		printf("Can't malloc for dst image!\n");
		return -1;
	}
	memset(dst_data,0,line_byte*height);
	MidX = width/2;
	MidY = height/2;
	
	for(y = 0; y < height; y+= 1)
	{
		dst_Imgindex = y*line_byte;
		for(x=0; x <width; x+= 1)
		{
			Yoffset = y - MidY;
			Xoffset = x - MidX;
			radian = atan2((double)Yoffset,(double)Xoffset);
			radius = sqrt((double)Xoffset*Xoffset+(double)Yoffset*Yoffset);
			radius = sqrt(radius)*degree;
			X = (int)(radius*cos(radian)+MidX);
			Y = (int)(radius*sin(radian)+MidY);
			if (X >= width) X = width-1;
			if (X < 0)	   X = 0;
			if (Y >= height)Y = height-1;
			if (Y < 0)	   Y = 0;
			src_Imgindex = Y*line_byte+3*X;
			
			dst_data[dst_Imgindex] = img_data[src_Imgindex];
			dst_data[dst_Imgindex+1] = img_data[src_Imgindex+1];
			dst_data[dst_Imgindex+2] = img_data[src_Imgindex+2];
			dst_Imgindex +=3;
		}
	}
	fwrite(dst_data,line_byte*height,1,Pinch_fp);
	fclose(Pinch_fp);
	Pinch_fp = NULL;
	free(dst_data);
	return 0;
}
```
