A=rand(256,64);

%将A分块
B=mat2cell(A,ones(256/16,1)*16,ones(64/16,1)*16);
%B{i,j}就是所要的分块矩阵

%将分块矩阵合并
C=cell2mat(B)
%C就是合并好的矩阵，即C=A
