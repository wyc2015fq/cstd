### **The wordcloud plot**

```
library(tm)
library(wordcloud2)
grandet <- readLines("grandet.txt")
corpus <- Corpus(VectorSource(grandet))
corpus1 <- tm_map(corpus,removeNumbers)
corpus1 <- tm_map(corpus1,removePunctuation)
corpus1 <- tm_map(corpus1,stripWhitespace)
corpus_tdm <- TermDocumentMatrix(corpus1)
corpus_mat <- as.matrix(corpus_tdm)
corpus_sum <- rowSums(corpus_mat)
dat <-    data.frame(cbind(term=names(corpus_sum),frequency=as.numeric(corpus_sum)))
dat$frequency <- as.numeric(dat$frequency)
wordcloud2(dat[dat$frequency>20,],backgroundColor = "black",shape = "R")
wordcloud2(dat[dat$frequency>20,],backgroundColor = "black",shape = 'star')
```
