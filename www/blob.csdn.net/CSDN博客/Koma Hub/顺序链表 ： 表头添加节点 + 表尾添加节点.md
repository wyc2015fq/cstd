在表尾添加节点：
```cpp
#include<stdio.h>
#include<stdlib.h>
typedef struct node{
        char value;
        struct node *next;
}node, *Linklist;
void main(){
        struct node *head = malloc(sizeof(struct node));
        struct node *tail = malloc(sizeof(struct node));
        scanf("%c",&head->value);
        head->next = tail;
        
        struct node *tmp;
        do{
                tmp = malloc(sizeof(struct node));
                scanf("%c",&tmp->value);
                tail->next = tmp;
                tail = tmp;
                tmp = NULL;
        }while(tail->value != 'q');
        
        tmp = head;
        do{
                printf("%c",tmp->value);
                tmp = tmp->next;
        }while(tmp->next != NULL);
        printf("\n");
}
```
