输出1~N的随机排列，如4，1，5，2，3

```cpp
#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+10;
int array[maxn];
void init(int n)
{
    for(int i=0;i<n;i++)
    array[i]=i+1;
}
void GetRandomArray(int number){
    for(int i=0;i<number;i++){
        int k=rand()%(number-i)+i;
        swap(array[i],array[k]);
    }
}

int main()
{
    freopen("test.in","w",stdout);
    srand((unsigned int)time(0));
    init(maxn);
    int T=100;
    cout<<T<<endl;
    while(T--){
        int n=2000;
        printf("%d\n",n);
        GetRandomArray(n);
        for(int i=0;i<n;i++)
            printf("%d%c",array[i],i==n-1?'\n':' ');
    }
    return 0;
}
```
