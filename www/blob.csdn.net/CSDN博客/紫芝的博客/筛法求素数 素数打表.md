#include<cstdio>
#include<cstring>
#include<cmath>

//素数打表 ，时间复杂度：O(nlog n) 
char vis[100000];
void prime(int n)
{
		int m=sqrt(n+0.5);
		memset(vis,0,sizeof(vis));
		
		// 1不是素数 
		for(int i=2;i<=m;i++)
		if(!vis[i])
		for(int j=i*i;j<=n;j+=i)
		vis[j]=1;//删除 		
}

int main()
{
	int n;
	while(scanf("%d",&n))
	{
		prime(n);
		//没有被删除的就是素数 
		for(int i=2;i<n;i++)
		if(!vis[i])
		printf("%d ",i);
		printf("\n");
	}
	return 0;
}
