Window{
    id:window;
    visible:true;
    width:400
    height:400
    ColumnLayout{
        anchors.fill:parent;
        spacing:0
        TabView{
            Layout.fillWidth:true
            Layout.preferredHeight:s.value

                title:"all"
            }
            Tab{
                title:"even"
            }
            Tab{
                title:"odd"
            }
            Tab{
                title:"multiple of 3"
            }
            Tab{
                title:"prime"
            }

        Layout.fillWidth:true;
        Layout.fillHeight:true;
        model:
            ListModel{
                ListElement{
                    name:"A";
                    title:"1";
                }
                ListElement{
                    name:"B";
                    title:"2";
                }
                ListElement{
                    name:"C";
                    title:"3";
                }
            }
            TableViewColumn{
                role:"name"
                title:"name"
            }
            TableViewColumn{
                role:"name"
                title:"number"
            }
        }
}
    Slider {
        id:s
        minimumValue:0;
        maximunValue:50
        anchors.bottom:parent.bottom
        Text{
            text:s.value.toFixed(0);

        }
    }
}
