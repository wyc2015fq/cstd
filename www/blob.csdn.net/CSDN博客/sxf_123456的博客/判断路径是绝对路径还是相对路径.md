"""
判断路径是绝对路径还是相对路径
"""
import os
def _get_sep(path):
    if isinstance(path,bytes):
        return b'/'
    else:
        return '/'
def isabs(path):
    s = os.fspath(path)  #判断path类型是否str或bytes,否抛出异常
    sep = _get_sep(s)
    return s.startswith(sep)
if __name__ == "__main__":
    path_1 = "root/test"
    path_2 = "./..test"
    print(isabs(path_1))
    print(isabs(path_2))
