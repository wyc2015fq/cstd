#include <stdio.h>
extern int b;
int a=b;
int b=10;
struct A
{
  int pA;     
};
class B
{
public:
      static A a;
      static int b[];
};
A B::a={*B::b};
int B::b[]={1,2,3};
int main()
{
    printf("%d,%d",a,B::a.pA);
    getchar();
    return 0;
}
输出 10,1
注意上边红色加亮的地方
那是由于在全局里的，全局变量的　初始化不是按顺序来的
