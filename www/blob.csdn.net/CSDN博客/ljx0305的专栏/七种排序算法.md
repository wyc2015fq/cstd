#include<stdio.h>
void CreatHeap(int a[],int n,int h)
{   int i,j,flag,temp;
     i=h;
  j=2*i+1;
  temp=a[i];
  flag=0;
  while(j<n && flag!=1)
  {  if(j<n-1 && a[j]<a[j+1]) j++;
      if(temp>a[j])
    flag=1;
   else
   {  a[i]=a[j];
      i=j;
   j=2*i+1;
   }
  }
   a[i]=temp;
}
void InitCreatHeap(int a[],int n)
{   int i;
    for(i=(n-1)/2;i>=0;i--)
  CreatHeap(a,n,i);
}
void HeapSort(int a[],int n)
{   int i,temp;
    InitCreatHeap(a,n);
for(i=n-1;i>0;i--)
{  temp=a[0];
    a[0]=a[i];
    a[i]=temp;
    CreatHeap(a,i,0);
}
}
main()
{  int n,i,a[100];
   printf("请问你要输入几个排序数:/n");
   scanf("%d",&n);
   printf("请输入你要排序的数值:/n");
   for(i=0;i<n;i++)
    scanf("%d",&a[i]);
   HeapSort(a,n);
   printf("排序后的:/n");
   for(i=0;i<n;i++)
    printf("%d/t",a[i]);
}

#include<stdio.h>
void QuickSort(int a[],int low,int high){
int i=low,j=high;
int temp=a[low];
while(i<j)
  {
   while(j>i&&temp<=a[j]) 
    j--;
      if(j>i)
   {
    a[i]=a[j];
        i++;
   }
   while(j>i&&a[i]<temp)
    i++;
   if(j>i)
   {
         a[j]=a[i];
   j--;
   }
  }
  a[i]=temp;
  if(low<i) QuickSort(a,low,i-1);
  if(i<high)QuickSort(a,j+1,high);
}

#include<stdio.h>
void ShellSort(int a[],int n,int d[],int numOfD)
{   int i,j,k,m,span,temp;
    for(m=0;m<=numOfD;m++)
{  span=d[m];
    for(k=0;k<span;k++)
    {  for(i=k;i<n-span;i=i+span)
       {
     temp=a[i+span];
     j=i;
     while(j>-1&&temp<=a[j])
     {  a[j+span]=a[j];
        j=j-span;
     }
     a[j+span]=temp;
    }
    }
}
}

{int a[100], b[10],n,i,k,j;

k=0;

   }while(j>0);
     ShellSort(a,n,b,k);

   printf("%d/t",a[i]);
