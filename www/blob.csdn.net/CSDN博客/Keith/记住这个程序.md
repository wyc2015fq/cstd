#include<iostream>
using namespace std;
int main()
{
	int number[]={1,2,3,4,5,6,7};
	char name[]={'y','i','n','p','e','n','g'};
	
	int *pNumber;
	char *pName;

	pNumber=number;
	pName=name;
	for(int i = 0; i < 7;i++)
	{
		cout<<*pNumber<<" at "<< pNumber<<endl;
		pNumber++;
	}
	for(int j = 0; j < 7;j++)
	{
		cout<<*pName<<" at "<< pName<<endl;
		pName++;
	}
	return 0;
}
